const names = ["ali", "sara", "omer", "lina", "mohammed","radwan"];

function longNames(names) {

    let result = [];

    for (let name of names) {
        if (name.length > 4) {
            result.push(name);
        }
    }

    return result;
}


// second Method 


function longNames2(names) {
    return names.filter(name => name.length > 4);
}
console.log(longNames(names));
